    <script>
        // --- 1. Three.js Background Logic ---
        const initThreeJS = () => {
            const container = document.getElementById('canvas-container');
            const scene = new THREE.Scene();
            
            const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.z = 30;

            const renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);
            container.appendChild(renderer.domElement);

            // Create Heart Shape
            const heartShape = new THREE.Shape();
            const x = 0, y = 0;
            heartShape.moveTo(x + 2.5, y + 2.5);
            heartShape.bezierCurveTo(x + 2.5, y + 2.5, x + 2.0, y, x, y);
            heartShape.bezierCurveTo(x - 3.0, y, x - 3.0, y + 3.5, x - 3.0, y + 3.5);
            heartShape.bezierCurveTo(x - 3.0, y + 5.5, x - 1.0, y + 7.7, x + 2.5, y + 9.5);
            heartShape.bezierCurveTo(x + 6.0, y + 7.7, x + 8.0, y + 5.5, x + 8.0, y + 3.5);
            heartShape.bezierCurveTo(x + 8.0, y + 3.5, x + 8.0, y, x + 5.0, y);
            heartShape.bezierCurveTo(x + 3.5, y, x + 2.5, y + 2.5, x + 2.5, y + 2.5);

            const geometry = new THREE.ShapeGeometry(heartShape);
            const material = new THREE.MeshBasicMaterial({ color: 0xdb2777, side: THREE.DoubleSide, transparent: true, opacity: 0.8 });

            const hearts = [];
            const heartCount = 25;

            for (let i = 0; i < heartCount; i++) {
                const heart = new THREE.Mesh(geometry, material.clone());
                // Randomize color slightly
                const hue = Math.random() > 0.5 ? 0.9 : 0.83; // Pink or Purpleish
                heart.material.color.setHSL(hue, 0.8, 0.6);
                
                // Random Position
                heart.position.set(
                    (Math.random() - 0.5) * 60,
                    (Math.random() - 0.5) * 40,
                    (Math.random() - 0.5) * 20
                );
                
                // Random Scale
                const s = Math.random() * 0.1 + 0.05;
                heart.scale.set(s, s, s);
                
                // Random Rotation
                heart.rotation.z = Math.PI; // Flip heart correctly
                heart.rotation.y = Math.random() * Math.PI;

                // Custom properties for animation
                heart.userData = {
                    speedY: Math.random() * 0.02 + 0.01,
                    speedRot: Math.random() * 0.01,
                    initialX: heart.position.x,
                    swaySpeed: Math.random() * 0.5 + 0.5,
                    timeOffset: Math.random() * 100
                };

                scene.add(heart);
                hearts.push(heart);
            }

            // Light
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
            scene.add(ambientLight);

            // Animation Loop
            let finaleMode = false;
            window.finaleMode = finaleMode; // Expose to window

            const animate = function (time) {
                requestAnimationFrame(animate);

                hearts.forEach(h => {
                    if (window.finaleMode) {
                        // Finale Animation: Float up and away fast
                        h.position.y += 0.2;
                        h.position.x += (h.position.x > 0 ? 0.1 : -0.1);
                        h.material.opacity -= 0.005;
                        h.rotation.y += 0.05;
                    } else {
                        // Idle Animation
                        h.position.y += h.userData.speedY;
                        h.rotation.y += h.userData.speedRot;
                        
                        // Gentle sway
                        h.position.x = h.userData.initialX + Math.sin(time * 0.001 * h.userData.swaySpeed + h.userData.timeOffset) * 2;

                        // Loop vertically
                        if (h.position.y > 25) {
                            h.position.y = -25;
                        }
                    }
                });

                renderer.render(scene, camera);
            };

            animate();

            // Resize Handler
            window.addEventListener('resize', () => {
                camera.aspect = window.innerWidth / window.innerHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(window.innerWidth, window.innerHeight);
            });
        };

        // --- 2. Navigation Logic (GSAP) ---
        let currentStep = 1;
        const totalSteps = 5;

        function nextStep(targetStep) {
            if (targetStep > totalSteps) return;

            // Update Progress Bar
            const progress = ((targetStep - 1) / (totalSteps - 1)) * 100;
            document.getElementById('progress-bar').style.width = `${progress}%`;

            // GSAP Transition
            const tl = gsap.timeline();
            
            // Exit Current Step
            const currentEl = document.getElementById(`step-${currentStep}`);
            tl.to(currentEl, {
                duration: 0.5,
                opacity: 0,
                y: -50,
                scale: 0.9,
                ease: "power2.in",
                onComplete: () => {
                    currentEl.style.display = 'none';
                    currentEl.classList.remove('active');
                }
            });

            // Enter Next Step
            const nextEl = document.getElementById(`step-${targetStep}`);
            nextEl.style.display = 'flex'; // Prepare for flex layout
            
            tl.fromTo(nextEl, 
                { opacity: 0, y: 50, scale: 0.9, visibility: 'visible' },
                { 
                    duration: 0.6, 
                    opacity: 1, 
                    y: 0, 
                    scale: 1, 
                    ease: "back.out(1.7)",
                    onStart: () => {
                        nextEl.classList.add('active');
                    }
                }
            );

            // Stagger children of next step
            tl.from(nextEl.children, {
                y: 20,
                opacity: 0,
                duration: 0.4,
                stagger: 0.1
            }, "-=0.4");

            currentStep = targetStep;
        }

        // --- 3. Polaroid Tilt Effect ---
        const polaroid = document.querySelector('.polaroid');
        const polaroidWrapper = document.getElementById('polaroid-container');

        polaroidWrapper.addEventListener('mousemove', (e) => {
            const rect = polaroidWrapper.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            
            const centerX = rect.width / 2;
            const centerY = rect.height / 2;
            
            const rotateX = ((y - centerY) / centerY) * -15; // Max 15deg rotation
            const rotateY = ((x - centerX) / centerX) * 15;

            polaroid.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale(1.05)`;
        });

        polaroidWrapper.addEventListener('mouseleave', () => {
            polaroid.style.transform = `rotateX(0deg) rotateY(0deg) rotate(3deg)`; // Reset to default "tossed" look
        });

        // --- 4. The Grand Finale ---
        function triggerFinale() {
            const btn = document.getElementById('celebrate-btn');
            const finalMsg = document.getElementById('final-message');

            // 1. Disable and fade out button
            gsap.to(btn, {
                duration: 0.5,
                opacity: 0,
                pointerEvents: 'none',
                y: 20
            });

            // 2. Show Final Message
            finalMsg.classList.remove('hidden');
            gsap.fromTo(finalMsg, 
                { opacity: 0, y: 30 },
                { delay: 0.5, duration: 1, opacity: 1, y: 0, ease: "power3.out" }
            );

            // 3. Trigger 3D Hearts floating away
            window.finaleMode = true;

            // 4. Confetti Explosion Sequence
            const duration = 5 * 1000;
            const animationEnd = Date.now() + duration;
            const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 100 };

            function randomInRange(min, max) {
                return Math.random() * (max - min) + min;
            }

            // Burst 1: Big Explosion
            confetti({
                particleCount: 150,
                spread: 100,
                origin: { y: 0.8 },
                colors: ['#db2777', '#9333ea', '#ffffff']
            });

            // Interval for continuous rain
            const interval = setInterval(function() {
                const timeLeft = animationEnd - Date.now();

                if (timeLeft <= 0) {
                    return clearInterval(interval);
                }

                const particleCount = 50 * (timeLeft / duration);

                // Rain of hearts and sparkles
                confetti(Object.assign({}, defaults, { 
                    particleCount, 
                    origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 },
                    shapes: ['heart'],
                    colors: ['#FFC0CB', '#FF69B4', '#FF1493']
                }));
                confetti(Object.assign({}, defaults, { 
                    particleCount, 
                    origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 },
                    shapes: ['circle'],
                    colors: ['#FFD700', '#FFFFFF']
                }));
            }, 250);
        }

        // Initialize Everything
        window.onload = () => {
            // GSAP Init for Step 1
            gsap.from("#step-1", { duration: 1, opacity: 0, y: 30, delay: 0.2 });
            initThreeJS();
        };

    </script>
